# Plan 2.1: Member CRUD (manager side, mobile-first)

## Objective

Ship the manager's member management surface: list with live search (D-06), create/edit forms with photo upload (D-01 to D-05), and a read-only profile view with recent activity. The list works on a 375px phone (cards) and a desktop (table). The form is mobile-first, single column, with sticky save and circular photo capture (D-02).

By the end of this plan: a manager can view, search, create, edit, and soft-delete (mark inactive) members; profile photos are uploaded to local storage with a circular preview and camera capture on mobile; every write is audited (AUDIT-01, PERF-08); the manager `/home` is updated with a Members count card linking here; the sidebar nav gains the new links (UI-SPEC §4.11); and the storage symlink is verified or created.

**Requirements covered:** MEM-01, MEM-02, MEM-03, MEM-04, MEM-07, MEM-08, MEM-09, PERF-04, PERF-11, PERF-12, AUDIT-01 (Member model), PERF-08 (Member model), PERF-13 (pint clean on every commit).

```php
class StoreMemberRequest extends FormRequest

            'status' => ['required', Rule::in(MemberStatus::ALL)],
            'leaving_date' => ['nullable', 'date', 'required_if:status,former'],
```

```php
        return $user && ($user->hasRole('admin') || $user->hasRole('super-admin'));

    public function rules(): array
```

### Task 5: Create the MemberController

**File:** `app/Http/Controllers/Mess/MemberController.php` (new)

```php
<?php

namespace App\Http\Controllers\Mess;

use App\Http\Controllers\Controller;
use App\Http\Requests\Mess\StoreMemberRequest;
use App\Http\Requests\Mess\UpdateMemberRequest;
use App\Models\Member;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Storage;
use Illuminate\View\View;

class MemberController extends Controller
{
    public function index(Request $request): View
    {
        $query = Member::query()
            ->orderByRaw("CASE status WHEN 'active' THEN 0 WHEN 'inactive' THEN 1 WHEN 'former' THEN 2 ELSE 3 END")
            ->orderBy('name');

        if ($search = trim((string) $request->query('q', ''))) {
            $query->where(function ($q) use ($search) {
                $q->where('name', 'like', "%{$search}%")
                    ->orWhere('mobile', 'like', "%{$search}%")
                    ->orWhere('email', 'like', "%{$search}%")
                    ->orWhere('room_or_seat', 'like', "%{$search}%");
            });
        }

        $members = $query->paginate(50)->withQueryString();
        $activeCount = Member::where('status', 'active')->count();

        return view('mess.members.index', compact('members', 'activeCount', 'search'));
    }

    public function create(): View
    {
        return view('mess.members.create', ['member' => new Member()]);
    }

    public function store(StoreMemberRequest $request): RedirectResponse
    {
        $data = $request->validated();
        $photo = $data['photo'] ?? null;
        unset($data['photo']);

        $member = Member::create($data);

        if ($photo) {
            $this->storePhoto($member, $photo);
        }

        return redirect()
            ->route('mess.members.show', $member)
            ->with('success', __('Member :name added.', ['name' => $member->name]));
    }

    public function show(Member $member): View
    {
        $member->load(['mealEntries' => fn ($q) => $q->latest('date')->limit(30)]);
        $member->load(['mealOffRequests' => fn ($q) => $q->latest('requested_at')->limit(10)]);
        $member->load(['guestMeals' => fn ($q) => $q->latest('date')->limit(10)]);

        return view('mess.members.show', compact('member'));
    }

    public function edit(Member $member): View
    {
        return view('mess.members.edit', compact('member'));
    }

    public function update(UpdateMemberRequest $request, Member $member): RedirectResponse
    {
        $data = $request->validated();
        $photo = $data['photo'] ?? null;
        unset($data['photo']);

        $member->update($data);

        if ($photo) {
            $this->storePhoto($member, $photo);
        }

        return redirect()
            ->route('mess.members.show', $member)
            ->with('success', __('Member :name updated.', ['name' => $member->name]));
    }

    public function destroy(Member $member): RedirectResponse
    {
        // Soft-delete semantics: mark inactive (NOT actually delete, to preserve
        // historical meal entries, expenses, payments). MEM-02 says "soft-delete (mark inactive)".
        $member->update(['status' => 'inactive']);
        // Preserve leaving_date? Optional. For v1, mark inactive without leaving_date.

        return redirect()
            ->route('mess.members.index')
            ->with('success', __('Member :name marked as inactive.', ['name' => $member->name]));
    }

    private function storePhoto(Member $member, \Illuminate\Http\UploadedFile $photo): void
    {
        $ext = $photo->getClientOriginalExtension();
        $path = "photos/{$member->id}.{$ext}";

        if ($member->photo_path) {
            Storage::disk('public')->delete($member->photo_path);
        }

        Storage::disk('public')->putFileAs(
            dirname($path),
            $photo,
            basename($path),
        );

        $member->update(['photo_path' => $path]);
    }
}
```

```php
class MemberSearchController extends Controller

    public function __invoke(Request $request): View
    {
        $query = Member::query()
            ->orderByRaw("CASE status WHEN 'active' THEN 0 WHEN 'inactive' THEN 1 WHEN 'former' THEN 2 ELSE 3 END")

        $search = trim((string) $request->query('q', ''));

        return view('mess.members._list', compact('members', 'activeCount', 'search'));
```
