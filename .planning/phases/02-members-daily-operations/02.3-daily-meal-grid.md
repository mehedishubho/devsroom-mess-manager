# Plan 2.3: Daily Meal Grid (Bulk Entry, Presets, Navigation, Meal-Off Aware)

## Objective

Ship the manager's daily meal grid at `/mess/meals`: rows = active members, columns = breakfast/lunch/dinner checkboxes (D-07, MEAL-01 to MEAL-11), with date navigation (◀ Today ▶ + date picker, D-08), preset buttons ("Mark all 3 meals" / "Mark all 0 meals", D-11/D-12), per-row quick actions (D-11), meal-off row disable (D-10), and a single "Save all changes" button that upserts all rows in one transaction (D-07, MEAL-08).

By the end of this plan: a manager navigates to any date, sees a grid of active members with the current meal state (pre-filled from existing entries), checks/unchecks meals across the grid, applies presets or per-row quick actions, and saves. Save persists all rows in one DB transaction (MEAL-08). The grid loads in < 100ms with 50 members (MEAL-11, PERF-04). Members with approved meal off for the date are shown grayed out with an "On meal off until MM-DD" badge and disabled checkboxes (D-10, MEAL-07). A unique key on `(mess_id, member_id, date)` is added to the `meal_entries` table to support the upsert (PITFALL #1).

**Requirements covered:** MEAL-01, MEAL-02, MEAL-03, MEAL-04, MEAL-05, MEAL-06, MEAL-07, MEAL-08, MEAL-09, MEAL-10, MEAL-11, GUEST-01, GUEST-02 (charge calculation uses MealType constants), PERF-04, PERF-11, PERF-12, AUDIT-01 (MealEntry model), PERF-08 (MealEntry model), PERF-13 (pint clean).

### Task 3: Create the MealGridService

**File:** `app/Services/MealGridService.php` (new)

The service builds the grid data: which active members to show, what their current meal entry is (if any), and which members are on approved meal off for the date. It also handles the bulk save (D-07, MEAL-08).

```php
<?php

namespace App\Services;

use App\Models\MealEntry;
use App\Models\MealOffRequest;
use App\Models\Member;
use App\Support\MemberStatus;
use App\Support\MealOffStatus;
use Carbon\Carbon;
use Illuminate\Support\Facades\DB;

class MealGridService
{
    /**
     * Build the grid data for a given date.
     *
     * @return array{members: \Illuminate\Support\Collection, date: Carbon, mealOffByMember: array}
     */
    public function buildGridData(Carbon $date): array
    {
        $activeMembers = Member::query()
            ->where('status', MemberStatus::ACTIVE)
            ->orderBy('name')
            ->get();

        $entries = MealEntry::query()
            ->where('date', $date->toDateString())
            ->whereIn('member_id', $activeMembers->pluck('id'))
            ->get()
            ->keyBy('member_id');

        $mealOffByMember = $this->approvedMealOffForDate($date, $activeMembers->pluck('id'));

        $rows = $activeMembers->map(function (Member $member) use ($entries, $mealOffByMember) {
            $entry = $entries->get($member->id);
            $off = $mealOffByMember[$member->id] ?? null;

            return (object) [
                'member' => $member,
                'breakfast' => $entry?->breakfast ?? false,
                'lunch' => $entry?->lunch ?? false,
                'dinner' => $entry?->dinner ?? false,
                'entry_id' => $entry?->id,
                'meal_off_until' => $off?->to_date,
                'editable' => $off === null,
            ];
        });

        return [
            'members' => $rows,
            'date' => $date,
            'mealOffByMember' => $mealOffByMember,
        ];
    }

    /**
     * Bulk save: upsert all rows in a single transaction.
     *
     * @param array<int, array{breakfast: bool, lunch: bool, dinner: bool, member_id: int}> $entries
     */
    public function bulkSave(Carbon $date, array $entries): void
    {
        $userId = auth()->id();
        $messId = \App\Models\Mess::activeId();
        $editableMembers = $this->editableMemberIdsForDate($date);

        DB::transaction(function () use ($date, $entries, $userId, $messId, $editableMembers) {
            foreach ($entries as $entry) {
                $memberId = (int) ($entry['member_id'] ?? 0);

                if (! in_array($memberId, $editableMembers, true)) {
                    continue; // Skip members on meal off (D-10, MEAL-07)
                }

                MealEntry::updateOrCreate(
                    [
                        'mess_id' => $messId,
                        'member_id' => $memberId,
                        'date' => $date->toDateString(),
                    ],
                    [
                        'breakfast' => (bool) ($entry['breakfast'] ?? false),
                        'lunch' => (bool) ($entry['lunch'] ?? false),
                        'dinner' => (bool) ($entry['dinner'] ?? false),
                        'entered_by' => $userId,
                    ]
                );
            }
        });
    }

    /**
     * @return array<int, MealOffRequest> Keyed by member_id
     */
    private function approvedMealOffForDate(Carbon $date, \Illuminate\Support\Collection $memberIds): array
    {
        return MealOffRequest::query()
            ->where('status', MealOffStatus::APPROVED)
            ->where('from_date', '<=', $date->toDateString())
            ->where('to_date', '>=', $date->toDateString())
            ->whereIn('member_id', $memberIds)
            ->get()
            ->keyBy('member_id')
            ->all();
    }

    /**
     * @return int[] Member IDs that can be edited on this date (no approved meal off)
     */
    private function editableMemberIdsForDate(Carbon $date): array
    {
        return Member::query()
            ->where('status', MemberStatus::ACTIVE)
            ->whereNotIn('id', function ($q) use ($date) {
                $q->select('member_id')
                    ->from('meal_off_requests')
                    ->where('status', MealOffStatus::APPROVED)
                    ->where('from_date', '<=', $date->toDateString())
                    ->where('to_date', '>=', $date->toDateString());
            })
            ->pluck('id')
            ->all();
    }
}
```

namespace App\Http\Requests\Mess;

use Illuminate\Foundation\Http\FormRequest;

class BulkSaveMealEntriesRequest extends FormRequest
{
    public function authorize(): bool
    {
        $user = $this->user();

        return $user && ($user->hasRole('admin') || $user->hasRole('super-admin'));
    }

    public function rules(): array
    {
        return [
            'date' => ['required', 'date'],
            'entries' => ['required', 'array'],
            'entries.*.member_id' => ['required', 'integer', 'exists:members,id'],
            'entries.*.breakfast' => ['nullable', 'boolean'],
            'entries.*.lunch' => ['nullable', 'boolean'],

}

### Task 5: Create the MealGridController

**File:** `app/Http/Controllers/Mess/MealGridController.php` (new)

```php
<?php

namespace App\Http\Controllers\Mess;

use App\Http\Controllers\Controller;
use App\Http\Requests\Mess\BulkSaveMealEntriesRequest;
use App\Services\MealGridService;
use Carbon\Carbon;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\View\View;

class MealGridController extends Controller
{
    public function __construct(
        private readonly MealGridService $gridService,
    ) {}

    public function index(Request $request): View
    {
        $date = $request->query('date')
            ? Carbon::parse($request->query('date'))->setTimezone(config('app.timezone'))
            : Carbon::now(config('app.timezone'));

        $data = $this->gridService->buildGridData($date);

        return view('mess.meals.index', $data);
    }

    public function store(BulkSaveMealEntriesRequest $request): RedirectResponse
    {
        $date = Carbon::parse($request->validated('date'))->setTimezone(config('app.timezone'));
        $this->gridService->bulkSave($date, $request->validated('entries'));

        return redirect()
            ->route('mess.meals.index', ['date' => $date->toDateString()])
            ->with('success', __('Meals saved for :date.', ['date' => $date->format('d M Y')]));
    }
}
```
