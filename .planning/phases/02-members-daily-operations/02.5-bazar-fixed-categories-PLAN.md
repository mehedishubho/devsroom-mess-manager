# Plan 2.5: Bazar + Fixed Expenses + Expense Categories

## Objective

Ship the manager's expense entry surfaces: bazar form (date, category filtered to bazar, purchased-by member, vendor, description, amount, optional receipt), fixed form (date, category filtered to fixed, description, amount), expense category manager (list + add + delete with default lock). The schema reconciliation (D-20) is applied: `kind` enum column added to `expense_categories`, `expense_type` column dropped from `expenses`. Default categories are seeded (D-21, CAT-01). Receipts use the same photo pattern as members (D-19).

By the end of this plan: a manager can record bazar and fixed expenses via separate forms (D-22 — category dropdown filtered by kind), view a list of all expenses, manage expense categories (create custom, see defaults are locked), and seed defaults run on `db:seed`. The `Expense` and `ExpenseCategory` models get the `Auditable` trait (AUDIT-01).

**Requirements covered:** BAZAR-01, BAZAR-02, BAZAR-03, BAZAR-04, BAZAR-05 (basic view), BAZAR-06 (no-receipt badge in list), FIXED-01, FIXED-02, FIXED-03, FIXED-04, CAT-01, CAT-02, CAT-03, CAT-04, PERF-04, PERF-11, PERF-12, AUDIT-01 (Expense + ExpenseCategory), PERF-08 (Expense + ExpenseCategory), PERF-13 (pint clean).

```php
<?php

namespace App\Http\Requests\Mess;

use App\Support\ExpenseKind;
use Illuminate\Foundation\Http\FormRequest;
use Illuminate\Validation\Rule;

class StoreExpenseRequest extends FormRequest
{
    public function authorize(): bool
    {
        $user = $this->user();

        return $user && ($user->hasRole('admin') || $user->hasRole('super-admin'));
    }

    public function rules(): array
    {
        $messId = \App\Models\Mess::activeId();
        $kind = $this->input('kind'); // bazar or fixed

        return [
            'kind' => ['required', Rule::in([ExpenseKind::BAZAR, ExpenseKind::FIXED])],
            'date' => ['required', 'date', 'before_or_equal:today'],
            'expense_category_id' => [
                'required', 'integer',
                Rule::exists('expense_categories', 'id')->where('mess_id', $messId)->where('kind', $kind),
            ],
            'purchased_by' => ['nullable', 'integer', 'exists:members,id'],
            'vendor' => ['nullable', 'string', 'max:255'],
            'description' => ['nullable', 'string', 'max:1000'],
            'amount' => ['required', 'numeric', 'min:0.01', 'max:99999999.99'],
            'receipt' => ['nullable', 'file', 'image', 'mimes:jpg,jpeg,png', 'max:5120'],
        ];
    }

    public function messages(): array
    {
        return [
            'amount.min' => __('Amount must be greater than zero.'),
            'date.before_or_equal' => __('Date cannot be in the future.'),
            'receipt.max' => __('Receipt must be 5 MB or smaller.'),
        ];
    }
}
```

### Task 10: Create the ExpenseService

**File:** `app/Services/ExpenseService.php` (new)

```php
<?php

namespace App\Services;

use App\Models\Expense;
use App\Models\ExpenseCategory;
use App\Models\Mess;
use Illuminate\Http\UploadedFile;
use Illuminate\Support\Facades\Storage;

class ExpenseService
{
    public function create(array $data, ?UploadedFile $receipt): Expense
    {
        $data['mess_id'] = Mess::activeId();
        $data['entered_by'] = auth()->id();
        $data['expense_category_id'] = $data['expense_category_id'] ?? null;
        unset($data['receipt']);

        $expense = Expense::create($data);

        if ($receipt) {
            $this->storeReceipt($expense, $receipt);
        }

        return $expense;
    }

    private function storeReceipt(Expense $expense, UploadedFile $receipt): void
    {
        $ext = $receipt->getClientOriginalExtension();
        $path = "receipts/{$expense->id}.{$ext}";

        Storage::disk('public')->putFileAs(dirname($path), $receipt, basename($path));
        $expense->update(['receipt_path' => $path]);
    }
}
```

**Acceptance criteria:**
- `ExpenseService::create` creates an expense and stores the receipt (if any).

```php
<?php

namespace App\Http\Controllers\Mess;

use App\Http\Controllers\Controller;
use App\Http\Requests\Mess\StoreExpenseRequest;
use App\Models\Expense;
use App\Models\ExpenseCategory;
use App\Models\Member;
use App\Services\ExpenseService;
use App\Support\ExpenseKind;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\View\View;

class ExpenseController extends Controller
{
    public function __construct(
        private readonly ExpenseService $service,
    ) {}

    public function index(Request $request): View
    {
        $query = Expense::query()
            ->with(['category', 'purchasedByMember'])
            ->orderBy('date', 'desc')
            ->orderBy('id', 'desc');

        if ($kind = $request->query('kind')) {
            $query->whereHas('category', fn ($q) => $q->where('kind', $kind));
        }

        $expenses = $query->paginate(20)->withQueryString();

        return view('mess.expenses.index', compact('expenses'));
    }

    public function createBazar(): View
    {
        $categories = ExpenseCategory::where('kind', ExpenseKind::BAZAR)->orderBy('sort_order')->orderBy('name')->get();
        $members = Member::where('status', 'active')->orderBy('name')->get();

        return view('mess.expenses.bazar.create', compact('categories', 'members'));
    }

    public function storeBazar(StoreExpenseRequest $request): RedirectResponse
    {
        $data = $request->validated();
        $receipt = $data['receipt'] ?? null;

        $expense = $this->service->create($data, $receipt);

        return redirect()
            ->route('mess.expenses.index')
            ->with('success', __('Bazar expense of ৳:amount saved.', ['amount' => number_format($expense->amount, 2)]));
    }

    public function createFixed(): View
    {
        $categories = ExpenseCategory::where('kind', ExpenseKind::FIXED)->orderBy('sort_order')->orderBy('name')->get();

        return view('mess.expenses.fixed.create', compact('categories'));
    }

    public function storeFixed(StoreExpenseRequest $request): RedirectResponse
    {
        $data = $request->validated();
        $receipt = $data['receipt'] ?? null;
        // Fixed expenses don't have purchased_by or vendor (per CONTEXT D-22, FIXED-01)
        $data['purchased_by'] = null;
        $data['vendor'] = null;
        $data['receipt'] = null; // no receipt for fixed

        $expense = $this->service->create($data, null);

        return redirect()
            ->route('mess.expenses.index')
            ->with('success', __('Fixed expense of ৳:amount saved.', ['amount' => number_format($expense->amount, 2)]));
    }
}
```
