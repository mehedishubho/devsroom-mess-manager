   public function bucket(\Carbon\Carbon $from, \Carbon\Carbon $to): array
   {
       $days = $from->diffInDays($to);
       if ($days <= 60)  return ['granularity' => 'day',   'step' => '1 day'];
       if ($days <= 365) return ['granularity' => 'week',  'step' => '1 week'];
